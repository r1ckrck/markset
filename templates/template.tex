% ---- Headings --------------------------------------------------------------
% H1 (section) — title + full-width rule. The rule is nudged up to sit close
% to the baseline; 0.55em is a text-relative nudge that scales with the
% heading font size.
\titleformat{\section}
  {\fontsize{\msfonthone}{\msfonthonelh}\selectfont\bfseries\color{textprimary}}
  {}{0pt}{}
  [\vspace{-0.55em}\rule{\textwidth}{\msrulethin}]
\titlespacing*{\section}{0pt}{\mshonebefore}{\mshoneafter}

  {\fontsize{\msfonthtwo}{\msfonthtwolh}\selectfont\bfseries\color{textprimary}}
